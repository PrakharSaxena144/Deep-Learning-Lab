P1 Perceptron (Binary Classification)

Key idea: augment X with bias column; update w += y*x when misclassified.

Perceptron Class
class Perceptron:
    def __init__(self, epochs=50):
        self.epochs = epochs

    def fit(self, X, y):
        # Prepend 1s for bias
        X = np.hstack((np.ones((X.shape[0], 1)), X))
        self.w = np.random.uniform(-1, 1, X.shape[1])
        for _ in range(self.epochs):
            for i in range(len(X)):
                if y[i] * np.dot(self.w, X[i]) <= 0:
                    self.w += y[i] * X[i]  # update rule

    def predict(self, X):
        X = np.hstack((np.ones((X.shape[0], 1)), X))
        return np.where(X @ self.w >= 0, 1, -1)

XOR Network (hardcoded weights)
def step(z): return 1 if z >= 0 else 0

def xor(x):
    # OR: weights=[1,1] bias=-0.5
    or_out  = step(np.dot(x, [1,1]) - 0.5)
    # NAND: weights=[-1,-1] bias=1.5
    nand_out= step(np.dot(x, [-1,-1]) + 1.5)
    # AND those two outputs: weights=[1,1] bias=-1.5
    return step(np.dot([or_out,nand_out], [1,1]) - 1.5)
P2 MLP for Boolean Functions

Hidden layer = identity matrix (eye(4)). Output perceptron trained on top. Two variants: step threshold vs sigmoid.

Step (Threshold) Perceptron
class StepPerceptron:
    def __init__(self):
        self.w = np.random.uniform(-1, 1, 4)
        self.b = np.random.uniform(-1, 1)

    def predict(self, x):
        return 1 if np.dot(self.w, x) + self.b >= 0 else 0

def train_step(truth, epochs=10):
    p = StepPerceptron()
    y = np.array(truth)
    H = np.eye(4)          # hidden outputs
    for _ in range(epochs):
        for x, t in zip(H, y):
            yh = p.predict(x)
            if yh != t:
                p.w += (t - yh) * x
                p.b += (t - yh)
Sigmoid Perceptron
class SigPerceptron:
    def __init__(self):
        self.w = np.random.uniform(-1, 1, 4)
        self.b = np.random.uniform(-1, 1)

    def sig(self, z):
        return 1 / (1 + np.exp(-z))

def train_sig(truth, epochs=10):
    p = SigPerceptron()
    y = np.array(truth)
    H = np.eye(4)
    for _ in range(epochs):
        for x, t in zip(H, y):
            yh = p.sig(np.dot(p.w, x) + p.b)
            p.w += (t - yh) * x  # always update (no threshold)
            p.b += (t - yh)
P3 Gradient Descent on f(x,y)

f(x,y) = x² + y² + 10·sin(x) + 10·cos(y) — manually derive ∂f/∂x and ∂f/∂y.

GD Function
def f(x, y):
    return x**2 + y**2 + 10*np.sin(x) + 10*np.cos(y)

def gradient_descent(x0, y0, lr=0.05, iters=100):
    x, y = float(x0), float(y0)
    path = [(x, y, f(x, y))]
    for _ in range(iters):
        dfdx = 2*x + 10*np.cos(x)   # ∂f/∂x
        dfdy = 2*y - 10*np.sin(y)   # ∂f/∂y
        x -= lr * dfdx
        y -= lr * dfdy
        path.append((x, y, f(x, y)))
    return np.array(path)   # shape (iters+1, 3)

P4 1-Hidden-Layer NN · make_moons · 4 Activations

Architecture: 2→48→1(sigmoid). Loss: Binary Cross-Entropy. Train with mini-batch GD (batch=64, lr=0.01, 10k iters).

Activations + Loss
def sigmoid(x): return 1/(1+np.exp(-x))

def activate(x, act):
    if act=="sigmoid": return sigmoid(x)
    if act=="tanh":    return np.tanh(x)
    if act=="relu":    return np.maximum(0,x)
    if act=="leaky":   return np.where(x>0,x,0.01*x)

def act_deriv(z, act):
    if act=="sigmoid": s=sigmoid(z); return s*(1-s)
    if act=="tanh":    return 1-np.tanh(z)**2
    if act=="relu":    return (z>0).astype(float)
    if act=="leaky":
        d=np.ones_like(z); d[z<0]=0.01; return d

def bce(y, yh):
    eps=1e-9
    return -np.mean(y*np.log(yh+eps)+(1-y)*np.log(1-yh+eps))
NN Class (2→48→1)
class NN:
    def __init__(self, act="relu", lr=0.01):
        self.act = act; self.lr = lr
        np.random.seed(0)
        self.W1 = np.random.randn(2,48)*0.1
        self.b1 = np.zeros((1,48))
        self.W2 = np.random.randn(48,1)*0.1
        self.b2 = np.zeros((1,1))

    def forward(self, X):
        self.Z1 = X @ self.W1 + self.b1
        self.A1 = activate(self.Z1, self.act)
        self.Z2 = self.A1 @ self.W2 + self.b2
        self.A2 = sigmoid(self.Z2)
        return self.A2

    def backward(self, X, y):
        m = X.shape[0]
        dZ2 = self.A2 - y
        dW2 = self.A1.T @ dZ2 / m
        db2 = np.sum(dZ2, axis=0, keepdims=True) / m
        dA1 = dZ2 @ self.W2.T
        dZ1 = dA1 * act_deriv(self.Z1, self.act)
        dW1 = X.T @ dZ1 / m
        db1 = np.sum(dZ1, axis=0, keepdims=True) / m
        self.W1 -= self.lr*dW1; self.b1 -= self.lr*db1
        self.W2 -= self.lr*dW2; self.b2 -= self.lr*db2

    def train(self, X, y, iters=10000, bs=64):
        losses = []
        for i in range(iters):
            idx = np.random.choice(len(X), bs)
            Xb, yb = X[idx], y[idx]
            yh = self.forward(Xb)
            if i % 200 == 0: losses.append(bce(yb, yh))
            self.backward(Xb, yb)
        return losses

    def predict(self, X):
        return (self.forward(X) > 0.5).astype(int)

Losses + Softmax
def relu(x):    return np.maximum(0,x)
def relu_d(x):  return (x>0).astype(float)

def softmax(x):
    e = np.exp(x - np.max(x, axis=1, keepdims=True))
    return e / np.sum(e, axis=1, keepdims=True)

def mse(y,yh):    return np.mean((y-yh)**2)
def mse_grad(y,yh): return (yh-y)/y.shape[0]

def cce(y,yh):
    return -np.mean(np.sum(y*np.log(yh+1e-9),axis=1))
def cce_grad(y,yh): return (yh-y)/y.shape[0]
Generic NN (variable layers)
class NN:
    def __init__(self, layers, lr=0.01):
        self.W=[]; self.b=[]
        np.random.seed(0)
        for i in range(len(layers)-1):
            self.W.append(np.random.randn(layers[i],layers[i+1])*0.1)
            self.b.append(np.zeros((1,layers[i+1])))
        self.lr = lr

    def forward(self, X):
        self.Z=[]; self.A=[X]
        for i in range(len(self.W)-1):     # hidden: ReLU
            z = self.A[-1] @ self.W[i] + self.b[i]
            self.Z.append(z); self.A.append(relu(z))
        z = self.A[-1] @ self.W[-1] + self.b[-1]  # output: softmax
        self.Z.append(z); self.A.append(softmax(z))
        return self.A[-1]

    def backward(self, y, loss_grad):
        dA = loss_grad
        for i in reversed(range(len(self.W))):
            dZ = dA if i==len(self.W)-1 else dA*relu_d(self.Z[i])
            self.W[i] -= self.lr * (self.A[i].T @ dZ)
            self.b[i] -= self.lr * np.sum(dZ, axis=0, keepdims=True)
            dA = dZ @ self.W[i].T
Train + Evaluate
def train(model, X, y, loss_fn, grad_fn, batch_size, epochs=100):
    losses=[]
    for _ in range(epochs):
        idx = np.random.permutation(len(X))
        X,y = X[idx], y[idx]
        for i in range(0,len(X),batch_size):
            xb,yb = X[i:i+batch_size], y[i:i+batch_size]
            pred = model.forward(xb)
            model.backward(yb, grad_fn(yb,pred))
        losses.append(loss_fn(y, model.forward(X)))
    return losses

P6 5 Optimizers from Scratch

All use the same forward/backward pass. Differ only in the update rule. Works for any parameter dict.

SGD & Momentum
# SGD
for k in params:
    params[k] -= lr * grads[k]

# Momentum (init: v={k:0 for k in params})
beta = 0.9
for k in params:
    v[k] = beta*v[k] - lr*grads[k]
    params[k] += v[k]
RMSprop & Adagrad
# Adagrad (init: s={k:0 for k in params})
for k in params:
    s[k] += grads[k]**2
    params[k] -= lr * grads[k] / (np.sqrt(s[k]) + 1e-8)

# RMSprop
beta = 0.9
for k in params:
    s[k] = beta*s[k] + (1-beta)*grads[k]**2
    params[k] -= lr * grads[k] / (np.sqrt(s[k]) + 1e-8)
Adam (init: m,v={k:0}, t=1)
b1,b2,eps = 0.9, 0.999, 1e-8
for k in params:
    m[k] = b1*m[k] + (1-b1)*grads[k]
    v[k] = b2*v[k] + (1-b2)*grads[k]**2
    m_hat = m[k] / (1 - b1**t)   # bias correction
    v_hat = v[k] / (1 - b2**t)
    params[k] -= lr * m_hat / (np.sqrt(v_hat) + eps)
t += 1
💡 Pattern: Create an init_state(opt) that returns the right dict of zeros, then a step(opt, params, grads, state, lr) that applies the rule. Loop over param keys inside.

P8 Overfitting + Regularization (L1, L2, Dropout)

Use a small training set to force overfit, then compare: No_Reg / L1 / L2 / Dropout / combinations.

Forward with Dropout
def forward(X, W1,b1, W2,b2, dropout=0.0):
    Z1 = X @ W1 + b1
    A1 = relu(Z1)
    mask = None
    if dropout > 0:
        mask = (np.random.rand(*A1.shape) > dropout) / (1-dropout)
        A1 *= mask          # inverted dropout (scale at train time)
    Z2 = A1 @ W2 + b2
    A2 = sigmoid(Z2)
    return Z1, A1, Z2, A2, mask
Loss with L1 + L2
def compute_loss(y, yh, W1, W2, l1=0, l2=0):
    m = y.shape[0]
    bce = -np.mean(y*np.log(yh+1e-8)+(1-y)*np.log(1-yh+1e-8))
    l1_t = l1*(np.sum(np.abs(W1))+np.sum(np.abs(W2)))/m
    l2_t = l2*(np.sum(W1**2)+np.sum(W2**2))/m
    return bce + l1_t + l2_t
Backward with Reg Gradients
def backward(X,y,Z1,A1,A2,W1,W2,mask,l1=0,l2=0):
    m = X.shape[0]
    dZ2 = A2 - y
    dW2 = (A1.T @ dZ2)/m
    db2 = np.sum(dZ2,axis=0,keepdims=True)/m
    dA1 = dZ2 @ W2.T
    if mask is not None: dA1 *= mask   # apply dropout mask
    dZ1 = dA1 * relu_d(Z1)
    dW1 = (X.T @ dZ1)/m
    db1 = np.sum(dZ1,axis=0,keepdims=True)/m
    # add reg terms to weight gradients
    dW2 += (l2*W2 + l1*np.sign(W2))/m
    dW1 += (l2*W1 + l1*np.sign(W1))/m
    return dW1,db1,dW2,db2

# Macro precision, recall, F1
def macro_metrics(y_true, y_pred):
    classes = np.unique(y_true)
    p_list, r_list = [], []
    for c in classes:
        tp = np.sum((y_pred==c)&(y_true==c))
        fp = np.sum((y_pred==c)&(y_true!=c))
        fn = np.sum((y_pred!=c)&(y_true==c))
        p_list.append(tp/(tp+fp+1e-9))
        r_list.append(tp/(tp+fn+1e-9))
    p,r = np.mean(p_list), np.mean(r_list)
    f1  = 2*p*r/(p+r+1e-9)
    return p, r, f1

Key Formulas Quick-Ref
# Sigmoid:         1/(1+e^-z)
# Sigmoid deriv:   s*(1-s)
# Tanh:            np.tanh(z)
# Tanh deriv:      1 - tanh(z)²
# ReLU:            max(0,z)
# ReLU deriv:      (z>0)
# Leaky ReLU:      z if z>0 else α*z  (α=0.01)
# BCE:            -mean(y*log(yh)+(1-y)*log(1-yh))
# CCE:            -mean(sum(y*log(yh), axis=1))
# MSE:             mean((y-yh)²)
# Softmax:         e^z / sum(e^z)  (subtract max for stability)
# L1 grad:        sign(W)
# L2 grad:        2*W  (or just W when λ already halved)
# Dropout mask:   (rand > p) / (1-p)  — inverted dropout
